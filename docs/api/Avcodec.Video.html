Module Avcodec.Video

module Video: sig .. end

Video codecs.

type id = Codec_id.video

Video codec ids

val get_name : id -> string

Return the name of the codec.

val find_id : string -> id

Return the id of a codec from his name.

  • Raises Failure if the codec is not found or is not a video codec.
val find_best_pixel_format : id -> Pixel_format.t

Return the best pixel format of a codec.

  • Raises Failure if the codec has no best pixel format.
val get_id : video Avcodec.t -> id

Return the id of the codec.

val get_width : video Avcodec.t -> int

Returns the width set for the codec.

val get_height : video Avcodec.t -> int

Returns the height set for the codec.

val get_sample_aspect_ratio : video Avcodec.t -> rational

Returns the sample aspect ratio set for the codec.

val get_pixel_format : video Avcodec.t -> Pixel_format.t

Returns the pixel format set for the codec.

val get_bit_rate : video Avcodec.t -> int

Returns the bit rate set for the codec.

val create_parser : id -> video Avcodec.Packet.parser

Avcodec.Video.create_parser id create an video packet parser.

  • Raises Failure if the parser creation failed.
val create_decoder : id -> video Avcodec.decoder

Avcodec.Video.create_decoder id create a video decoder.

  • Raises Failure if the decoder creation failed.
val create_encoder : ?bit_rate:int ->
       ?frame_rate:int -> id -> video Avcodec.encoder

Avcodec.Video.create_encoder ~bit_rate:bit_rate id create a video encoder.

  • Raises Failure if the encoder creation failed.
